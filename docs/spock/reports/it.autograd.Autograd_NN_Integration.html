).setRqsGradient(true)
def W2 = Tsr.of(
        [[ 1.8095],
         [-0.4269],
         [-1.1110]]
).setRqsGradient(true)
def W1s = []
def z1s = []
def hiddenResults = []
def W2s = []

def forwardAndBackward = ( Tsr x ) ->
{
    W1s.add(W1.toString())
    def z1 = x.dot(W1)
    z1s.add(z1.toString())
    def hidden = sig(z1)
    hiddenResults.add(hidden.toString())
    W2s.add(W2.toString())
    def z2 = hidden.dot(W2)
    z2s.add(z2.toString())
    def pred = sig(z2)
    outputResults.add(pred.toString())
    def error = (y - pred)
    errors.add(error.toString())
    def loss = (error**2).mean()
    losses.add(loss.toString())
    pred.backward(error)
    W1.applyGradient()
    W2.applyGradient()
    return pred
}
