package neureka.utility;

public class Decider
{
    public interface Path { boolean go(Object o); }

    public static Path check(Class[] classes, Path[] paths){
        return (o) -> {
            for (Class c : classes){
                if(c.isInstance(o)){
                    for(Path a : paths) {
                        if(a.go(o)) return true;
                    }
                }
            }
            return false;
        };
    }

    public void go(Object o){
        if(
            check(
                    new Class[]{},
                    new Path[]{
                        check(
                                new Class[]{},
                                new Path[]{}
                        )
                    }
            ).go(o)
        ){
        }
    }
}
